%  -- add parentheses in fraction espression if there is a superscript. 
 \makeatletter
 \let\old@frac=\frac
 \def\frac#1#2{%
 	\@ifnextchar^
 	{\left(\old@frac{#1}{#2}\right)}
 	{\old@frac{#1}{#2}}%
 }
 
 \let\old@dfrac=\dfrac
 \def\dfrac#1#2{%
 	\@ifnextchar^
 	{\left(\old@dfrac{#1}{#2}\right)}
 	{\old@dfrac{#1}{#2}}%
 }
